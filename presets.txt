    if (action == "refresh"){
        socket.emit("refresh","", (response) => {
            if (response == null){
                statusb.style = "color: red;";
                statusb.style = "color: red;";
                statusb.textContent = "Aucun nouveaux commentaires.Essayez de reafraichir plus tard.";

                c1.innerHTML = response;

        });
    }else{
        statusb.style = "color: red;";

        `;
    }
}
